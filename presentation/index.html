Class methods (prototype)

Constructor function is called every time the instance is created. It's ok for instance attributes, but not for methods


var SomeClass = function (name){
    this.instanceMethod = function () {};
};

SomeClass.prototype.classMethod = function () {};

var obj1 = new SomeClass(),
    obj2 = new SomeClass();

obj1.instanceMethod === obj2.instanceMethod // false
obj1.classMethod === obj2.classMethod // true

Event emitter implementation


App.EventEmitter = function () {
    this._events = {};
};

App.EventEmitter.prototype.on = function (event, fn, ctx) {
    if (typeof this._events[event] === 'undefined') {
        this._events[event] = [];
    }
    this._events[event].push({fn: fn, ctx: ctx || window);
};

Event emitter implementation


App.EventEmitter.prototype.emit = function (event, data) {
    if (typeof this._events[event] !== 'undefined') {
        var i = 0,
            n = this._events[event].length;
        for (i; i < n; ++i) {
            this._events[event][i].fn.call(this._events[event][i].ctx, data);
        }
    }
};
