Get form values

There is no way to send the FormData via WebSockets ...


	var self = this;

	document.querySelector('.add-tweet-form', function (event) {
	    event.preventDefault();
	    var form = event.target,
	        data = {
	            title: form.elements['tweet-title'],
	            msg: form.elements['tweet-msg']
	        };
	    self.fire('addtweet', data);
	});

Send data in real-time

... so we have to parse the data to JSON string


	view.on('addtweet', function (data) {
	    data = JSON.stringify(data);
	    exampleSocket.send(data);
	});

Add a tweet to the list

Controller


	exampleSocket.onmessage = function (event) {
	    model.addItem(event.data);
	};

	model.on('additem', function (data) {
	    view.addTweet(data);
	});

View


	var addTweet = function (data) {
	    var listEl = document.querySelector('.tweets-list'),
	        tweetEl = this.render(data);
	    if (tweetEl) {
	        listEl.appendChild(tweetEl);
	    }
	};

Removing tweets

React on buttons clicks

We use the event delegation


	var self = this,
	    list = document.querySelector('.tweets-list');

	list.addEventListener('click', function (event) {
	    var target = event.target;
	    if (target.tagName === 'LI') {
	        self.fire('removetweet', target.dataset.id);
	    }
	});

Send data to the server


	view.on('removetweet', function (id) {
	    exampleSocket.send(id);
	});

Remove a tweet from list

Controller


	exampleSocket.onmessage = function (event) {
	    model.removeItem(event.data);
	};

	model.on('removeitem', function (id) {
	    view.removeTweet(id);
	});

View


	var removeTweet = function (id) {
	    var listEl = document.querySelector('.tweets-list'),
	        tweetEl = listEl.querySelector('[data-id="' + id + '"]');
	    if (tweetEl) {
	        listEl.removeChild(tweetEl);
	    }
	};
